} async function loadArticles() { const grid = document.getElementById('articles-grid'); const allArticles = []; let loaded = 0; for (const feed of FEEDS) { try { const res = await fetch(PROXY + encodeURIComponent(feed.url)); allArticles.push({ ...item, source: feed.source, relevant }); }); loaded++; } catch(e) { console.warn('Flux indisponible:', feed.source, e.message); } } // Trier : articles pertinents en premier, puis par date allArticles.sort((a, b) => { if (a.relevant && !b.relevant) return -1;
